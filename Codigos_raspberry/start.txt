#!/bin/sh -e
#
# rc.local
#
# Este script se ejecuta al final de cada nivel de ejecución multiusuario.
# Asegúrate de que termine con "exit 0" en caso de éxito.

# Mostrar la IP local
_IP=$(hostname -I) || true
if [ "$_IP" ]; then
  printf "My IP address is %s\n" "$_IP"
fi

# Ejecutar detector de caídas
cd Desktop/Proyecto-microprocesadores1/Codigos_raspberry
python3 receptor_dual_ble.py

exit 0

[Unit]
Description=/etc/rc.local Compatibility
ConditionPathExists=/etc/rc.local
After=network.target

[Service]
Type=forking
ExecStart=/etc/rc.local start
TimeoutSec=0
RemainAfterExit=yes
SysVStartPriority=99

[Install]
WantedBy=multi-user.target
